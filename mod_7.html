    with open('reporte_ventas.txt', 'w') as reporte:

üìà Ejercicio 3: Generador de Gr√°ficos de Texto

Objetivo: Crear una funci√≥n que genere gr√°ficos de barras usando caracteres ASCII.

Instrucciones:

  • Lee datos de "datos_graficos.txt"
  • Cada l√≠nea tiene formato: "Etiqueta:Valor"
  • Genera un gr√°fico de barras horizontal
  • Guarda el gr√°fico en "grafico.txt"
‚úÖ Soluci√≥n
def generar_grafico(archivo_datos, archivo_salida):
        datos = {}
        
        # Leer datos
        with open(archivo_datos, 'r') as archivo:
            for linea in archivo:
                linea = linea.strip()
                if ':' in linea:
                    etiqueta, valor = linea.split(':')
                    datos[etiqueta.strip()] = int(valor.strip())
        
        # Generar gr√°fico
        with open(archivo_salida, 'w') as grafico:
            grafico.write("GR√ÅFICO DE BARRAS\n")
            grafico.write("=" * 30 + "\n\n")
            
            for etiqueta, valor in datos.items():
                barra = "‚ñà" * valor
                grafico.write(f"{etiqueta:12} |{barra} {valor}\n")

    # Uso de la funci√≥n
    generar_grafico('datos_graficos.txt', 'grafico.txt')
    print("Gr√°fico generado exitosamente!")
